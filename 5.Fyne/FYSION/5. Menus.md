Example 
Open dir function which will be opened inside the menu
```Go
// stateful
type gui struct {
	win       fyne.Window
	directory *widget.Label
}

```
```Go
func (g *gui) openProject() {
	dialog.ShowFolderOpen(func(dir fyne.ListableURI, err error) {
		if err != nil {
			dialog.ShowError(err, g.win)
			return
		}
		if dir == nil {
			return
		}
		name := dir.Name()
		log.Println("dir opened", dir)
		g.win.SetTitle("Fysion App" + name)
	}, g.win)

}

```
### And the menu 
```Go
func (g *gui) makeMenu() *fyne.MainMenu {
	file := fyne.NewMenu("File",
		fyne.NewMenuItem("Open Project", g.openProject))
	return fyne.NewMainMenu(file)
}
```
